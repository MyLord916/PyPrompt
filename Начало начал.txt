                                        # Ввод и вывод через консоль
s = 'Python'  # Присвоение строки 'Python' в переменную <s>

print(s)                                                                           # Вывод в консоль значение строки <s>

print(s, s, s)                           # передавая несколько аргументов по умолчанию они будут выводиться через пробел

print(s, s, sep='\a')                            # намеренно задаем разделитель нескольких аргументов в виде символа <*>

print(s, end='*')          # задаем разделитель между следующим выводом текста, по умолчанию это перенос на новую строку

        # Внутри конструкции <print()> есть управляющие конструкции выполняемые в строке:
        # \n - Переход на новую строчку
        # \t - Отступ табуляции
        # \a - Воспроизводит звук системного динамика (при его наличии)
        # А так же символ <\> позволяет вывести в сообщении спец символы и изолировать их от выполнением кода
        # например: \\, \", \''''

input(s)       # Запрашивает ввод от пользователя и возвращает его, опционально можно вывести сообщение <s> перед вводом
